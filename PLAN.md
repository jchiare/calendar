# Calendar Application Implementation Plan

## Executive Summary

This plan outlines the architecture and implementation strategy for a modern calendar application built with **Next.js 16 (App Router)**, **Tailwind CSS**, and **Convex** as the backend/database. The application will feature multiple calendar views, integrations with major calendar providers (Google, Outlook, Apple), and an AI-powered chat assistant for intelligent scheduling.

---

## Tech Stack

- **Frontend**: Next.js 16 (App Router), React, Tailwind CSS
- **Backend/Database**: Convex
- **Authentication**: Clerk
- **AI**: OpenAI API
- **External Integrations**: Google Calendar API, Microsoft Graph API, CalDAV (Apple)

## 3. Database Schema (Convex)

```typescript
// convex/schema.ts
import { defineSchema, defineTable } from "convex/server";
import { v } from "convex/values";

export default defineSchema({
  // Users table - stores user profile and preferences
  users: defineTable({
    clerkId: v.string(),
    email: v.string(),
    name: v.optional(v.string()),
    imageUrl: v.optional(v.string()),
    preferences: v.object({
      defaultView: v.union(v.literal("day"), v.literal("week"), v.literal("month")),
      weekStartsOn: v.number(), // 0 = Sunday, 1 = Monday
      timeZone: v.string(),
      defaultEventDuration: v.number(), // in minutes
      workingHours: v.object({
        start: v.string(), // "09:00"
        end: v.string(),   // "17:00"
      }),
      theme: v.union(v.literal("light"), v.literal("dark"), v.literal("system")),
    }),
    createdAt: v.number(),
    updatedAt: v.number(),
  })
    .index("by_clerk_id", ["clerkId"])
    .index("by_email", ["email"]),

  // Calendars table - user's calendars (both local and synced)
  calendars: defineTable({
    userId: v.id("users"),
    name: v.string(),
    description: v.optional(v.string()),
    color: v.string(),
    type: v.union(
      v.literal("local"),
      v.literal("google"),
      v.literal("outlook"),
      v.literal("apple")
    ),
    externalId: v.optional(v.string()),
    integrationId: v.optional(v.id("integrations")),
    syncEnabled: v.boolean(),
    lastSyncAt: v.optional(v.number()),
    syncToken: v.optional(v.string()),
    isDefault: v.boolean(),
    isVisible: v.boolean(),
    createdAt: v.number(),
    updatedAt: v.number(),
  })
    .index("by_user", ["userId"])
    .index("by_user_and_type", ["userId", "type"])
    .index("by_integration", ["integrationId"])
    .index("by_external_id", ["externalId", "type"]),

  // Events table - calendar events
  events: defineTable({
    userId: v.id("users"),
    calendarId: v.id("calendars"),
    title: v.string(),
    description: v.optional(v.string()),
    location: v.optional(v.string()),
    startTime: v.number(),
    endTime: v.number(),
    isAllDay: v.boolean(),
    timeZone: v.string(),
    recurrenceRule: v.optional(v.string()),
    recurrenceId: v.optional(v.id("events")),
    externalId: v.optional(v.string()),
    externalEtag: v.optional(v.string()),
    source: v.union(
      v.literal("local"),
      v.literal("google"),
      v.literal("outlook"),
      v.literal("apple"),
      v.literal("ai")
    ),
    status: v.union(
      v.literal("confirmed"),
      v.literal("tentative"),
      v.literal("cancelled")
    ),
    visibility: v.union(
      v.literal("public"),
      v.literal("private"),
      v.literal("confidential")
    ),
    reminders: v.array(v.object({
      type: v.union(v.literal("email"), v.literal("popup")),
      minutesBefore: v.number(),
    })),
    attendees: v.optional(v.array(v.object({
      email: v.string(),
      name: v.optional(v.string()),
      status: v.union(
        v.literal("accepted"),
        v.literal("declined"),
        v.literal("tentative"),
        v.literal("pending")
      ),
      isOrganizer: v.boolean(),
    }))),
    conferenceData: v.optional(v.object({
      type: v.union(v.literal("meet"), v.literal("teams"), v.literal("zoom")),
      url: v.string(),
    })),
    aiGenerated: v.optional(v.boolean()),
    aiSuggestionId: v.optional(v.id("chatMessages")),
    createdAt: v.number(),
    updatedAt: v.number(),
  })
    .index("by_user", ["userId"])
    .index("by_calendar", ["calendarId"])
    .index("by_user_and_time", ["userId", "startTime"])
    .index("by_calendar_and_time", ["calendarId", "startTime"])
    .index("by_external_id", ["externalId", "source"])
    .index("by_recurrence", ["recurrenceId"]),

  // Integrations table - OAuth connections
  integrations: defineTable({
    userId: v.id("users"),
    provider: v.union(
      v.literal("google"),
      v.literal("outlook"),
      v.literal("apple")
    ),
    accessToken: v.string(),
    refreshToken: v.optional(v.string()),
    tokenExpiresAt: v.optional(v.number()),
    appleAppPassword: v.optional(v.string()),
    appleUserId: v.optional(v.string()),
    status: v.union(
      v.literal("active"),
      v.literal("expired"),
      v.literal("revoked"),
      v.literal("error")
    ),
    lastError: v.optional(v.string()),
    scopes: v.array(v.string()),
    externalAccountId: v.optional(v.string()),
    externalAccountEmail: v.optional(v.string()),
    createdAt: v.number(),
    updatedAt: v.number(),
  })
    .index("by_user", ["userId"])
    .index("by_user_and_provider", ["userId", "provider"]),

  // Chat conversations table
  chatConversations: defineTable({
    userId: v.id("users"),
    title: v.optional(v.string()),
    contextType: v.optional(v.union(
      v.literal("general"),
      v.literal("scheduling"),
      v.literal("event_edit"),
      v.literal("availability")
    )),
    contextEventId: v.optional(v.id("events")),
    isActive: v.boolean(),
    createdAt: v.number(),
    updatedAt: v.number(),
  })
    .index("by_user", ["userId"])
    .index("by_user_active", ["userId", "isActive"]),

  // Chat messages table
  chatMessages: defineTable({
    conversationId: v.id("chatConversations"),
    userId: v.id("users"),
    role: v.union(v.literal("user"), v.literal("assistant"), v.literal("system")),
    content: v.string(),
    isStreaming: v.boolean(),
    streamingContent: v.optional(v.string()),
    toolCalls: v.optional(v.array(v.object({
      id: v.string(),
      name: v.string(),
      arguments: v.string(),
    }))),
    toolResults: v.optional(v.array(v.object({
      toolCallId: v.string(),
      result: v.string(),
    }))),
    actions: v.optional(v.array(v.object({
      type: v.union(
        v.literal("event_created"),
        v.literal("event_updated"),
        v.literal("event_deleted"),
        v.literal("availability_checked"),
        v.literal("suggestion_made")
      ),
      eventId: v.optional(v.id("events")),
      data: v.optional(v.string()),
    }))),
    createdAt: v.number(),
  })
    .index("by_conversation", ["conversationId"])
    .index("by_conversation_time", ["conversationId", "createdAt"]),

  // Sync queue table
  syncQueue: defineTable({
    userId: v.id("users"),
    integrationId: v.id("integrations"),
    calendarId: v.optional(v.id("calendars")),
    operation: v.union(
      v.literal("full_sync"),
      v.literal("incremental_sync"),
      v.literal("push_event"),
      v.literal("delete_event")
    ),
    eventId: v.optional(v.id("events")),
    externalEventId: v.optional(v.string()),
    status: v.union(
      v.literal("pending"),
      v.literal("processing"),
      v.literal("completed"),
      v.literal("failed")
    ),
    attempts: v.number(),
    lastError: v.optional(v.string()),
    scheduledFor: v.number(),
    processedAt: v.optional(v.number()),
    createdAt: v.number(),
  })
    .index("by_status", ["status"])
    .index("by_scheduled", ["status", "scheduledFor"])
    .index("by_user", ["userId"]),

  // Webhook subscriptions table
  webhookSubscriptions: defineTable({
    userId: v.id("users"),
    integrationId: v.id("integrations"),
    provider: v.union(v.literal("google"), v.literal("outlook")),
    subscriptionId: v.string(),
    resourceId: v.optional(v.string()),
    channelId: v.optional(v.string()),
    expiresAt: v.number(),
    createdAt: v.number(),
  })
    .index("by_integration", ["integrationId"])
    .index("by_subscription_id", ["subscriptionId"])
    .index("by_expiry", ["expiresAt"]),

  // Event embeddings for AI semantic search
  eventEmbeddings: defineTable({
    eventId: v.id("events"),
    userId: v.id("users"),
    embeddingText: v.string(),
    embedding: v.array(v.float64()),
    createdAt: v.number(),
  })
    .index("by_event", ["eventId"])
    .index("by_user", ["userId"])
    .vectorIndex("by_embedding", {
      vectorField: "embedding",
      dimensions: 1536,
      filterFields: ["userId"],
    }),
});
```
